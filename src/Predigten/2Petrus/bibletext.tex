\chapter{Allgemeines}
\section{Verfassungszeit}
\begin{itemize}
    \item ca. 67-68 \nChr
    \item unter Kaiser Nero
\end{itemize}
\section{Verfassungsort}
\begin{itemize}
    \item Rom, da der Brief kurz vor seinem Tod geschrieben wurde 1:13-14.
    \item Petrus wusste von seinem Tod \bibleverse{Joh}(21:18-19)
\end{itemize}
\section{Verfasser}
\begin{itemize}
    \item Der Apostel Petrus Vers 1:1
    \item Er ist der Verfasser eines zweiten Briefes. Der erste wird in Vers 3:1 erwähnt
\end{itemize}
\section{Grund}
\begin{itemize}
    \item Entlarven von Irrlehren
    \item Einen wachsenden Glauben Leben
    \item Erwartung auf die Wiederkunft Jesus
\end{itemize}
\section{Empfänger}
\begin{itemize}
    \item Da Petrus in Vers 3:1 erwähnt, dass dies der zweite Brief an sie ist, sind es die gleichen Empfänger \bibleverse{IPetr}(1:1)
\end{itemize}
